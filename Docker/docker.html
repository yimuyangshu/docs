常用命令 ​

shell
# 查看容器列表
docker ps    # 已启动
docker ps -a # 所有

# 容器状态
docker stats

# 启动/停止/重启指定容器
docker start/stop/restart <container-id>

# 进入指定容器的终端
docker exec -it <container-id> /bin/bash

# 改变容器重启策略 / 挂载卷 / 网络设置 / 资源限制(cpu、内存)
docker update --restart=always <container-id>
